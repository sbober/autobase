001    import liquibase.*;
002    import java.io.*;
003    import java.util.*;
004    import java.net.*;
005    import java.util.zip.*;
006    import org.apache.log4j.*;
007    
008    public class WarFileOpener extends FileSystemFileOpener implements FileOpener {
009    
010      private static final Logger logger = Logger.getLogger(WarFileOpener.class);
011    
012      public WarFileOpener() throws IOException {
013        super(expandStream(createZipStream()).getAbsolutePath());
014      }
015    
016      /**
017      * Expands the given stream and returns the base directory where files were expanded into.
018      */
019      private static File expandStream(final ZipInputStream source) throws IOException {
020        final String timestamp = Long.toString(System.currentTimeMillis(), 16);
021        final File base = new File(System.getProperty("java.io.tmpdir"), "autobase-" + timestamp).getAbsoluteFile();
022        base.mkdirs();
023        logger.debug("Base directory: " + base.getAbsolutePath());
024        deleteDirOnExit(base);
025        for(ZipEntry entry = source.getNextEntry(); entry != null; entry = source.getNextEntry()) {
026          final File entryFile = new File(base, entry.getName()).getAbsoluteFile();
027          logger.debug("Writing out file: " + entryFile.getAbsolutePath());
028          entryFile.getParentFile().mkdirs();
029          if(entry.isDirectory()) {
030            source.closeEntry();
031            entryFile.mkdir();
032          } else {
033            entryFile.createNewFile();
034            final OutputStream entryStream = new BufferedOutputStream(new FileOutputStream(entryFile));
035            for(int c = source.read(); c != -1; c = source.read()) {
036              entryStream.write(c);
037            }
038            source.closeEntry();
039            entryStream.close();
040          }
041        }
042        source.close();
043        return base;
044      }
045    
046      private static void deleteDirOnExit(final File base) {
047        final Runnable deleteBase = new Runnable() {
048          public void run() { 
049            delete(base);
050          }
051    
052          private void delete(final File dir) {
053            final File[] files = dir.listFiles();
054            if(files != null) {
055              for(final File file : files) {
056                delete(file);
057              }
058            }
059            dir.delete();
060          }
061        };
062        Runtime.getRuntime().addShutdownHook(new Thread(deleteBase));
063      }
064    
065      private static ZipInputStream createZipStream() throws IOException {
066        return new ZipInputStream(new BufferedInputStream(new ClassLoaderFileOpener().getResourceAsStream("/autobase.zip")));
067      }
068    
069    }
